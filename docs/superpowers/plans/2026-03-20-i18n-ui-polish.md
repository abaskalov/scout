- [ ] **Step 1: Create i18n hook and provider**

Create `dashboard/src/i18n/index.ts`:
```typescript
import { createContext, useContext, useState, useEffect, type ReactNode } from 'react';
import ru from './ru.json';
import en from './en.json';
import uz from './uz.json';

export type Locale = 'ru' | 'en' | 'uz';

const STORAGE_KEY = 'scout_locale';
const translations: Record<Locale, Record<string, string>> = { ru, en, uz };

function detectLocale(): Locale {
  const saved = localStorage.getItem(STORAGE_KEY);
  if (saved && saved in translations) return saved as Locale;
  const lang = navigator.language.slice(0, 2);
  if (lang in translations) return lang as Locale;
  return 'ru';
}

interface I18nContextValue {
  locale: Locale;
  setLocale: (l: Locale) => void;
  t: (key: string, params?: Record<string, string>) => string;
}

const I18nContext = createContext<I18nContextValue>(null!);

export function I18nProvider({ children }: { children: ReactNode }) {
  const [locale, setLocaleState] = useState<Locale>(detectLocale);

  function setLocale(l: Locale) {
    setLocaleState(l);
    localStorage.setItem(STORAGE_KEY, l);
  }

  function t(key: string, params?: Record<string, string>): string {
    let str = translations[locale][key] ?? translations.ru[key] ?? key;
    if (params) {
      for (const [k, v] of Object.entries(params)) {
        str = str.replace(`{${k}}`, v);
      }
    }
    return str;
  }

  return <I18nContext value={{ locale, setLocale, t }}>{children}</I18nContext>;
}

export function useTranslation() {
  return useContext(I18nContext);
}

export const LOCALE_MAP: Record<Locale, string> = {
  ru: 'ru-RU', en: 'en-US', uz: 'uz-Latn',
};
```

```typescript
type Locale = 'ru' | 'en' | 'uz';

const T: Record<Locale, Record<string, string>> = {
  ru: {
    'picker.hint': 'Нажмите на элемент с ошибкой',
    'picker.cancel': 'Отмена',
    'loading.screenshot': 'Создание скриншота...',
    'login.title': 'Scout — Вход',
    'login.subtitle': 'Войдите, чтобы сообщать о багах',
    // ... all ~30 widget strings
  },
  en: { /* English translations */ },
  uz: { /* Uzbek translations */ },
};

function detectLocale(): Locale {
  const lang = navigator.language.slice(0, 2);
  if (lang in T) return lang as Locale;
  return 'en';
}

let currentLocale: Locale = detectLocale();

export function t(key: string): string {
  return T[currentLocale]?.[key] ?? T.en[key] ?? key;
}
```
